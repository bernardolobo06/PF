a)  funA [2,3,5,1] (y=2; ys=[3,5,1])
    = 2^2 + funA [3,5,1] (y=3; ys=[5,1])
    = 2^2 + 3^2 + funA [5,1] (y=5; ys=[1])
    = 2^2 + 3^2 + 5^2 + funA [1] (y=1; ys=[])
    = 2^2 + 3^2 + 5^2 + 1^2 + funA []
    = 2^2 + 3^2 + 5^2 + 1^2 + 0
    = 4 + 9 + 25 + 1 + 0
    = 39

b)  funB [8,5,12] (h=8; t=[5,12])

    = [8] : [] : [12] : funB []
    = [8] : [] : [12] : []
    = [8,12]
